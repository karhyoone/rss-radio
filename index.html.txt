  <script>
    // Tailwind already loaded
    function toggleMobileMenu() {
      document.getElementById('mobileMenu').classList.toggle('hidden');
    }

    // Render Packages
    function renderPackages() {
      const data = JSON.parse(document.getElementById('packagesData').textContent);
      const grid = document.getElementById('packagesGrid');
      grid.innerHTML = '';
      
      data.forEach(pkg => {
        const card = document.createElement('div');
        card.className = 'bg-zinc-800 p-8 rounded-3xl border border-amber-500/20 hover:border-amber-400 transition flex flex-col';
        card.innerHTML = `
          <h3 class="text-2xl font-bold mb-2">${pkg.name}</h3>
          <div class="text-5xl font-bold text-amber-400 mb-6">$${pkg.price}</div>
          <p class="text-zinc-400 mb-8 flex-1">${pkg.desc}</p>
          <div class="flex flex-col gap-3">
            <a href="${pkg.paypal}" target="_blank" class="flex items-center justify-center gap-2 bg-blue-600 hover:bg-blue-700 text-white py-3 rounded-2xl font-medium transition">
              <i class="fa-brands fa-paypal"></i> Pay with PayPal
            </a>
            <a href="${pkg.card}" onclick="alert('Stripe checkout integration coming soon - contact admin to activate card payments')" class="flex items-center justify-center gap-2 bg-zinc-700 hover:bg-zinc-600 text-white py-3 rounded-2xl font-medium transition">
              <i class="fa-solid fa-credit-card"></i> Pay with Card
            </a>
            <a href="${pkg.crypto}" target="_blank" class="flex items-center justify-center gap-2 bg-emerald-600 hover:bg-emerald-700 text-white py-3 rounded-2xl font-medium transition">
              <i class="fa-solid fa-coins"></i> Pay with Crypto
            </a>
          </div>
        `;
        grid.appendChild(card);
      });
    }

    // Form handling
    document.getElementById('submitForm').addEventListener('submit', function(e) {
      e.preventDefault();
      this.classList.add('hidden');
      document.getElementById('successMessage').classList.remove('hidden');
      setTimeout(() => {
        this.reset();
        this.classList.remove('hidden');
        document.getElementById('successMessage').classList.add('hidden');
      }, 5000);
    });

    // Initialize
    window.onload = () => {
      renderPackages();
      const audio = document.getElementById('radioPlayer');
      if (audio) audio.volume = 0.7;
    };
  </script>
